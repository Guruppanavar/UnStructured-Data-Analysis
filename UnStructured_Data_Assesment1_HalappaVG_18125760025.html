docs_modi = modi['text'].str.lower().str.replace('[^a-z^#^@ ]', '')

def clean_sentence_modi(text):
    words_modi = text.split(' ')
    words_clean_modi = (stemmer_modi.stem(word_modi) for word_modi in words_modi if word_modi not in stopwords_modi)
    return ' '.join(words_clean_modi)
